// Saadni Firestore Security Rules - Complete v7 (Mar 14, 2026)
// This file fixes ALL permission denied errors and enables messaging, user management, and data sync

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // MARK: - Helper Functions

    function isAuth() {
      return request.auth != null;
    }

    function isAuthUser(userId) {
      return request.auth != null && request.auth.uid == userId;
    }

    function isConversationParticipant(conversationId) {
      return isAuth() && exists(/databases/$(database)/documents/conversations/$(conversationId)) &&
             request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participantIds;
    }

    // MARK: - Users Collection

    match /users/{userId} {
      // Users can read all user profiles (needed for chat participant names and provider info)
      allow read: if isAuth();

      // Users can only write/update their own profile
      allow create, update, delete: if isAuthUser(userId);
    }

    // MARK: - Services Collection

    match /services/{serviceId} {
      // Anyone (authenticated or not) can read services
      allow read: if isAuth();

      // Only the service provider can write/update/delete
      allow create: if isAuth();
      allow update, delete: if isAuth() && resource.data.providerId == request.auth.uid;
    }

    // Services -> Applications subcollection
    match /services/{serviceId}/applications/{applicationId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update, delete: if isAuth() &&
        (resource.data.userId == request.auth.uid ||
         get(/databases/$(database)/documents/services/$(serviceId)).data.providerId == request.auth.uid);
    }

    // MARK: - Conversations Collection

    match /conversations/{conversationId} {
      // Participants can read conversation info
      allow read: if isAuth() && isConversationParticipant(conversationId);

      // Anyone authenticated can create conversations
      allow create: if isAuth();

      // Participants can update their conversation
      allow update: if isAuth() && isConversationParticipant(conversationId);

      // Participants can delete conversation (archive/soft delete)
      allow delete: if isAuth() && isConversationParticipant(conversationId);

      // MARK: - Messages Subcollection (CRITICAL FIX)

      match /messages/{messageId} {
        // Conversation participants can read all messages
        allow read: if isConversationParticipant(conversationId);

        // Users can create messages in conversations they're part of
        allow create: if isAuth() && isConversationParticipant(conversationId);

        // Users can only update/delete their own messages
        allow update, delete: if isAuth() && resource.data.senderId == request.auth.uid;
      }

      // MARK: - Typing Indicators Subcollection (CRITICAL FIX)

      match /typingIndicators/{userId} {
        // Participants can read typing indicators
        allow read: if isConversationParticipant(conversationId);

        // Users can write their own typing status
        allow create, update: if isAuth() && isAuthUser(userId) && isConversationParticipant(conversationId);

        // Users can delete their own typing status
        allow delete: if isAuth() && isAuthUser(userId);
      }
    }

    // MARK: - Reviews Collection

    match /reviews/{reviewId} {
      // Users can read all reviews
      allow read: if isAuth();

      // Users can create reviews
      allow create: if isAuth();

      // Only the reviewer can update their review
      allow update: if isAuth() && resource.data.reviewerId == request.auth.uid;

      // Only the reviewer can delete their review
      allow delete: if isAuth() && resource.data.reviewerId == request.auth.uid;
    }

    // MARK: - Applications Collection

    match /applications/{applicationId} {
      // Applicants and service providers can read applications
      allow read: if isAuth() &&
        (resource.data.userId == request.auth.uid ||
         get(/databases/$(database)/documents/services/$(resource.data.serviceId)).data.providerId == request.auth.uid);

      // Users can create applications
      allow create: if isAuth();

      // Applicants and providers can update
      allow update: if isAuth() &&
        (resource.data.userId == request.auth.uid ||
         get(/databases/$(database)/documents/services/$(resource.data.serviceId)).data.providerId == request.auth.uid);

      // Applicants can delete (withdraw)
      allow delete: if isAuth() && resource.data.userId == request.auth.uid;
    }

    // MARK: - Transactions/Wallet Collection

    match /transactions/{transactionId} {
      // Users can only read their own transactions
      allow read: if isAuth() && resource.data.userId == request.auth.uid;

      // Only authenticated users can create (server-side only in production)
      allow create: if isAuth();

      // No updates or deletes to transactions (immutable ledger)
      allow update, delete: if false;
    }

    // MARK: - Denied by Default

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
